class Human {

<!-- constructor 함수 안에서 this.gender으로 지정하던 값을 gender으로만 지정할 수 있다. -->

gender = 'male';

    <!-- Arrow Function 사용 -->
    printGender = () => {

        <!-- 함수안에선 여전히 this 사용 -->

        console.log(this.gender)
    }

}

class Person extends Human {

    name = '강정욱'
    <!-- 상속받은 클래스의 값을 변경 할 때도 this. 를 제외해도됨 -->
    gender = 'female'

    printMyName = () => {
        console.log(this.name)
    }

}
